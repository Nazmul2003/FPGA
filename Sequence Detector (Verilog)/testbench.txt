`timescale 1ns / 1ps

module test1;
	// Inputs
	reg rst;
	reg clk;
	// Outputs
	wire f;
	wire q1;
	wire q0;
	wire [4:0] seq;
	// Instantiate the Unit Under Test (UUT)
	sequence_detector uut (
		.rst(rst), 
		.clk(clk), 
		.f(f), 
		.q1(q1), 
		.q0(q0),
		.seq(seq)
	);
	always begin
		#20 clk = ~clk;
	end
	initial begin
		// Initialize Inputs
		rst = 0;
		clk = 0;
		// Wait 1000 ns for global reset to finish
		#1000;
		$stop;
	end
endmodule
